package id.kaytrust.core.utils.nat.lib;

import java.math.BigInteger;

import org.web3j.abi.datatypes.Address;
import org.web3j.abi.datatypes.Bool;

import org.web3j.abi.datatypes.generated.Uint256;
import lombok.Getter;
import lombok.Setter;

import org.web3j.abi.datatypes.StaticStruct;

@Getter
@Setter
public class ProducType extends StaticStruct {
    private BigInteger price;

    private String recipient;

    private BigInteger stock;

    private Boolean enabled;

    public ProducType(BigInteger price, String recipient, BigInteger stock, Boolean enabled) {
        super(new Uint256(price),new Address(recipient),new Uint256(stock),new Bool(enabled));
        this.price = price;
        this.recipient = recipient;
        this.stock = stock;
        this.enabled = enabled;
    }

    public ProducType(Uint256 price, Address recipient, Uint256 stock, Bool enabled) {
        super(price,recipient,stock,enabled);
        this.price = price.getValue();
        this.recipient = recipient.getValue();
        this.stock = stock.getValue();
        this.enabled = enabled.getValue();
    }
}
