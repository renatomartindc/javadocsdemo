package id.kaytrust.core.utils.nat.lib;

import java.math.BigInteger;

import lombok.Getter;
import lombok.Setter;

@Setter
@Getter
public class ProductBean {
    
    private String addressIdentifier;

    private BigInteger price;

    private BigInteger stock;

    private String recipient;

    private Boolean enabled;

    public ProductBean(){ 
    }

    public ProductBean(String addressIdentifier, BigInteger price,  String recipient, BigInteger stock, Boolean enabled){
        this.addressIdentifier = addressIdentifier;
        this.price = price;
        this.recipient = recipient;
        this.enabled = enabled;
        this.stock = stock;
    }

    public ProductBean(String addressIdentifier, ProducType type){ 
        this.addressIdentifier = addressIdentifier;
        this.price = type.getPrice();
        this.recipient = type.getRecipient();
        this.enabled = type.getEnabled();
        this.stock = type.getStock();
    }
}
