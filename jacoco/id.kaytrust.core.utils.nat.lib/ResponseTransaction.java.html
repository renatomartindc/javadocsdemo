package id.kaytrust.core.utils.nat.lib;

import org.web3j.protocol.core.methods.response.EthSendTransaction;
import org.web3j.protocol.core.methods.response.TransactionReceipt;

import lombok.Getter;
import lombok.Setter;

@Setter
@Getter
public class ResponseTransaction {
    private EthSendTransaction ethSendTransaction;
    private TransactionReceipt receipt ;

    ResponseTransaction(EthSendTransaction ethSendTransaction, TransactionReceipt receipt){
        this.ethSendTransaction = ethSendTransaction;
        this.receipt = receipt;
    }
}
