package id.kaytrust.proof.util;

import java.util.HashMap;
import java.util.Map;

import id.kaytrust.core.utils.model.verification.ProofTypeBuilder;
import id.kaytrust.proof.EthereumAttestation2019Builder;
import id.kaytrust.proof.EthereumAttestationRegistryIntermediateStatus2021Builder;
import id.kaytrust.proof.EthereumAttestationSmartContractBuilder;

public class StrategyProof {
    private Map<String, ProofTypeBuilder> mapping;

    public StrategyProof(){
        mapping = new HashMap<>();
        mapping.put(EthereumAttestation2019Builder.TYPE, new EthereumAttestation2019Builder());
        mapping.put(EthereumAttestationRegistryIntermediateStatus2021Builder.TYPE, new EthereumAttestationRegistryIntermediateStatus2021Builder());
        mapping.put(EthereumAttestationSmartContractBuilder.TYPE, new EthereumAttestationSmartContractBuilder());
    }

    public ProofTypeBuilder getProof(String key){
        return mapping.get(key);
    }
}
