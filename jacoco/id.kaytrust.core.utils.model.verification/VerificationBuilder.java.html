package id.kaytrust.core.utils.model.verification;

import java.util.ArrayList;
import java.util.Date;
import java.util.List;
import java.util.Map;

import org.json.JSONException;
import org.json.JSONObject;

public class VerificationBuilder {

	VerificationBuilder(){}

	public static final String FORMAT_DATE_ISO = "yyyy-MM-dd'T'HH:mm:ssXXX";
	private static final String[] CONTEXT_CREDENTIAL = { "https://www.w3.org/2018/credentials/v1",
			"https://schema.org/" };

	public static <T> VerifiableCredential<T> build(T credentialSubject, String[] credentialType,
			 Date issuanceDate,  Date expirationDate, String issuer,  String id) {

		VerifiableCredential<T> verifiableCredential = new VerifiableCredential<>();

		verifiableCredential.setContext(CONTEXT_CREDENTIAL);
		
		verifiableCredential.setType(credentialType);

		verifiableCredential.setCredentialSubject(credentialSubject);

		verifiableCredential.setExpirationDate(expirationDate);

		verifiableCredential.setId(id);

		verifiableCredential.setIssuanceDate(issuanceDate != null ? issuanceDate : new Date());

		verifiableCredential.setIssuer(issuer);

		return verifiableCredential;
	}

	public static String addParameter(String json, Map<String, Object> extra) throws JSONException {

		JSONObject mainObject = new JSONObject(json);

		List<String> exclusiveId = new ArrayList<>();
		exclusiveId.add("id");
		exclusiveId.add("proof");
		exclusiveId.add("credetialSubject");
		exclusiveId.add("issuer");
		exclusiveId.add("type");

		for (Map.Entry<String, Object> entry : extra.entrySet()) {
			if (!(exclusiveId.contains(entry.getKey()))) {
				mainObject.put(entry.getKey(), entry.getValue());
			}
		}
		return mainObject.toString();
	}
}
