package id.kaytrust.core.utils.model.verification;

import java.io.IOException;

import org.web3j.protocol.exceptions.TransactionException;

import id.kaytrust.core.utils.exception.CapabilityException;
import id.kaytrust.core.utils.exception.MnidInvalidAddressException;
import id.kaytrust.core.utils.nat.IdentityNative;
import id.kaytrust.core.utils.nat.VerifyNative;
import id.kaytrust.proof.util.CheckProof;

public interface ProofTypeBuilder {

	<T> String generateProof(VerifiableCredential<T> vc, String issuer, IdentityNative identityNative,
			String privateKey, String addressContractDestination, double factor) throws MnidInvalidAddressException, IOException, TransactionException, CapabilityException ;

	ProofType getProofType();

	String getVc(String vc);

	default <T> String buildProof(VerifiableCredential<T> vc, String issuer, IdentityNative identityNative,
			String privateKey, String addressContractDestination, double factor) throws MnidInvalidAddressException, IOException, TransactionException, CapabilityException {
		vc.setProof(getProofType());
		return generateProof(vc, issuer, identityNative, privateKey,
				addressContractDestination, factor);
	}

	<T> String addProof(VerifiableCredential<T> vc, String netWorkId, String addressContract);

	CheckProof checkProof(VerifyNative verification, String addressFrom, String addressContract, String vc, String did)
			throws MnidInvalidAddressException, IOException;

}
