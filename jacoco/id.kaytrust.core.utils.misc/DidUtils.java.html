package id.kaytrust.core.utils.misc;

public class DidUtils {
    DidUtils(){}
    /**
	 * @param did             The <a target="_blank" href=
	 *                        "https://www.w3.org/2019/09/did-wg-charter.html#:~:text=Decentralized%20Identifiers%20(DIDs)%20are%20a,identity%20provider%2C%20or%20certificate%20authority.">Decentralized
	 *                        identifier</a>
     * @return the specific identifier from a did.
     */
    public static String getSpecificIdentifierFromDid(String did){
        return did.replaceFirst("(.*):","");
    }
}
