package id.kaytrust.core.identity.ethereum.admin;

import id.kaytrust.core.identity.base.IdentityManager;
import id.kaytrust.core.utils.exception.ParamEmptyException;

/**
 * The Generic Identity Manager interface with capabilities to create identities
 * and authorize devices. This interface is compliant with any ethereum network.
 */
public interface EthereumIdentityManager extends IdentityManager {
	/**
	 * Sets an instance to interact with the Identity Manager ethereum smart
	 * contract.
	 * 
	 * @param hostRpc         host and port where ethereum node serves RPC requests
	 * @param token           can be blank, is token field Authorization
	 * @param privateKey      The private key used to interact with the smart
	 *                        contract.
	 * @param contractAddress Contract Address where the Smart contract lives.
	 * @return instance of EthereumIdentityManager
	 * @throws ParamEmptyException      if a required parameter is empty.
	 */
	static EthereumIdentityManager instance(String hostRpc, String token, String privateKey, String contractAddress) throws ParamEmptyException {
		return new EthereumIdentityManagerImpl(hostRpc, token, privateKey, contractAddress);
	}
	
	/**
	 * Sets an instance to interact with the Identity Manager ethereum smart
	 * contract.
	 * 
	 * @param hostRpc         host and port where ethereum node serves RPC requests
	 * @param token           can be blank, is token field Authorization
	 * @param privateKey      The private key used to interact with the smart
	 *                        contract.
	 * @param contractAddress Contract Address Identity Manager.
	 * @param blockTime 	  time mining of block in milliseconds, it's optional by default it's 15000
	 * @throws ParamEmptyException if a required parameter is empty.
	 */
	static EthereumIdentityManager instance(String hostRpc, String token, String privateKey, String contractAddress, int blockTime) throws ParamEmptyException {
		return new EthereumIdentityManagerImpl(hostRpc, token, privateKey, contractAddress, blockTime);
	}
}
