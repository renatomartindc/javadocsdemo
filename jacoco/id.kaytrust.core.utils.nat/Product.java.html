package id.kaytrust.core.utils.nat;

import java.io.IOException;

import org.web3j.protocol.exceptions.TransactionException;

import id.kaytrust.core.utils.nat.lib.ProductBean;

public interface Product {
    /**
     * Create instance Product 
     * @param host node ethereum
     * @param token access token 
     * @param contractAddress address contract products
     * @param blockTime time of minning
     * @return instence product
     */
    public static Product build(String host, String token, String contractAddress, int blockTime) {
        return new ProductImpl(host, token, contractAddress, blockTime);
    }

    /**
     * get product by identifier
     * @param addressQuery who do query
     * @param addressProductIdentifier address of product
     * @return Product
     */
    ProductBean getProduct(String addressQuery, String addressProductIdentifier)throws IOException; 

    /**
     * add product
     * @param privateKey owner
     * @param product to insert
     * @param factor calculate gas
     * @return hash transaction
     * @throws IOException error network
     * @throws TransactionException error trasaction in blockchain
     */
    String addProduct(String privateKey, ProductBean product, double factor)throws IOException, TransactionException; 
    
    /**
     * Update product
     * @param privateKey owner
     * @param product to insert
     * @param factor calculate gas
     * @return hash transaction
     * @throws IOException error network
     * @throws TransactionException error trasaction in blockchain
     */
    String updateProduct(String privateKey, ProductBean product, double factor)throws IOException, TransactionException; 
    
    /**
     * add maintainer contract
     * @param privateKey owner
     * @param addressContract
     * @param factor calculate gas
     * @return hash transaction
     * @throws IOException error network
     * @throws TransactionException error trasaction in blockchain
     */
    String addLimitedProductMaintainer(String privateKey, String addressContract, double factor)throws IOException, TransactionException;
    
    /**
     * Returns Version of Network
     * @return version of network in hexa
     * @throws IOException error of net
     */
    String getNetVersion() throws IOException;

    /**
     * close the connection
     */
    void shutdown();
}
