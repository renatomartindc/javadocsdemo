package id.kaytrust.core.utils.nat;

import java.math.BigInteger;
import java.util.Arrays;
import java.util.Collections;

import org.web3j.abi.FunctionEncoder;
import org.web3j.abi.TypeReference;
import org.web3j.abi.datatypes.Address;
import org.web3j.abi.datatypes.Bool;
import org.web3j.abi.datatypes.Function;
import org.web3j.abi.datatypes.generated.Uint256;

import id.kaytrust.core.utils.nat.lib.ProducType;

public class ProductByteCode {

    ProductByteCode(){}

    private static final String FUNC_SETPRODUCT = "setProduct";

    private static final String FUNC_GETPRODUCT = "getProduct";

    public static final String FUNC_UPDATEPRODUCT = "updateProduct";

    public static final String FUNC_ADDLIMITEDPRODUCTMAINTAINER = "addLimitedProductMaintainer";

    public static String addProduct(BigInteger price, String productIdentifier, String recipient, BigInteger stock, Boolean enabled) {
        final Function function = new Function(
                FUNC_SETPRODUCT, 
                Arrays.asList(new Uint256(price), 
                new Address(160, productIdentifier), 
                new Address(160, recipient), 
                new Uint256(stock),
                new Bool(enabled)), 
                Collections.<TypeReference<?>>emptyList());
        return FunctionEncoder.encode(function);
    }
    
    public static String updateProduct(BigInteger price, String productIdentifier, String recipient, BigInteger stock, Boolean enabled) {
        final Function function = new Function(
                FUNC_UPDATEPRODUCT, 
                Arrays.asList(new Uint256(price), 
                new Address(160, productIdentifier), 
                new Address(160, recipient), 
                new Uint256(stock), 
                new Bool(enabled)), 
                Collections.<TypeReference<?>>emptyList());
                
                return FunctionEncoder.encode(function);
    }

    public static Function getProduct(String productIdentifier) {
        return new Function(FUNC_GETPRODUCT, 
                Arrays.asList(new Address(160, productIdentifier)), 
                Arrays.asList(new TypeReference<ProducType>() {}));
         
    }
 
    public static String addLimitedProductMaintainer(String addressContract) {
        final org.web3j.abi.datatypes.Function function = new org.web3j.abi.datatypes.Function(
                FUNC_ADDLIMITEDPRODUCTMAINTAINER, 
                Arrays.asList(new org.web3j.abi.datatypes.Address(160, addressContract)), 
                Collections.<TypeReference<?>>emptyList());
                return FunctionEncoder.encode(function);
    }


}
