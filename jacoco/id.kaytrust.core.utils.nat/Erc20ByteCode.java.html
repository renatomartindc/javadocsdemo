package id.kaytrust.core.utils.nat;

import java.math.BigInteger;
import java.util.Arrays;
import java.util.Collections;

import org.web3j.abi.FunctionEncoder;
import org.web3j.abi.TypeReference;
import org.web3j.abi.datatypes.Address;
import org.web3j.abi.datatypes.Function;
import org.web3j.abi.datatypes.Utf8String;
import org.web3j.abi.datatypes.generated.Uint256;
import org.web3j.abi.datatypes.generated.Uint8;

public class Erc20ByteCode {

    private static final String FUNC_BALANCEOF = "balanceOf";
    private static final String FUNC_MINT = "mint";
    private static final String FUNC_SYMBOL = "symbol";
    private static final String FUNC_TOTALSUPPLY = "totalSupply";
    private static final String FUNC_TRANSFER = "transfer";
    private static final String FUNC_TRANSFERFROM = "transferFrom";
    private static final String FUNC_DECIMALS = "decimals";
    private static final String FUNC_NAME = "name";
    private static final String FUNC_OWNER = "owner";
    private static final String FUNC_TRANSFEROWNERSHIP = "transferOwnership";
    private static final String FUNC_SETVALIDATIONRULESCONTRACTADDRESS = "setValidationRulesContractAddress";
    private static final String FUNC_GETVALIDATIONRULESCONTRACTADDRESS = "getValidationRulesContractAddress";
    private static final String FUNC_SETADDRESSASPRODUCTIDENTIFIER = "setAddressAsProductIdentifier";
    private static final String FUNC_APPROVE = "approve";
    private static final String FUNC_BURN = "burn";
    private static final String FUNC_BURNFROM = "burnFrom";

    private Erc20ByteCode() {
    }

    public static Function symbol() {
        
        return  new Function(
                FUNC_SYMBOL, 
                Arrays.asList(),
                Arrays.asList(new TypeReference<Utf8String>() {}));

    }


    public static Function totalSupply() {
        
        return new Function(
                FUNC_TOTALSUPPLY, 
                Arrays.asList(), 
                Arrays.asList(new TypeReference<Uint256>() {}));

    }


    public static Function balanceOf(String account) {
        
        return new Function(
                FUNC_BALANCEOF, 
                Arrays.asList(new Address(160, account)), 
                Arrays.asList(new TypeReference<Uint256>() {}));

    }


    public static Function decimals() { 
        
        return new Function(
                FUNC_DECIMALS, 
                Arrays.asList(), 
                Arrays.asList(new TypeReference<Uint8>() {}));
    }


    public static Function name() {
        
        return new Function(
                FUNC_NAME, 
                Arrays.asList(), 
                Arrays.asList(new TypeReference<Utf8String>() {}));

    }


    public static Function owner() { 
        
        return new Function(
                FUNC_OWNER, 
                Arrays.asList(), 
                Arrays.asList(new TypeReference<Address>() {}));

    }

    public static String transfer(String recipient, BigInteger amount) {
        
        final Function function = new Function(
                FUNC_TRANSFER, 
                Arrays.asList(new Address(160, recipient), 
                new Uint256(amount)), 
                Collections.emptyList());

        return FunctionEncoder.encode(function);
    }


    public static String transferFrom(String sender, String recipient, BigInteger amount) {
        
        final Function function = new Function(
                FUNC_TRANSFERFROM, 
                Arrays.asList(
                new Address(160, sender), 
                new Address(160, recipient), 
                new Uint256(amount)), 
                Collections.emptyList());

        return FunctionEncoder.encode(function);
    }

    public static String transferOwnership(String newOwner) {
        
        final Function function = new Function(
                FUNC_TRANSFEROWNERSHIP, 
                Arrays.asList(new Address(160, newOwner)), 
                Collections.emptyList());

        return FunctionEncoder.encode(function);
    }


    public static String mint(String to, BigInteger amount) {
        
        final Function function = new Function(
                FUNC_MINT, 
                Arrays.asList(new Address(160, to),
                new Uint256(amount)),
                Collections.emptyList());

        return FunctionEncoder.encode(function);
    }


    public static String setContractValidationRules(String addressContractRules) {
        final Function function = new Function(
                FUNC_SETVALIDATIONRULESCONTRACTADDRESS, 
                Arrays.asList(new Address(160, addressContractRules)), 
                Collections.emptyList());
        
        return FunctionEncoder.encode(function);
    }

    public static String setProductIdentifier(String addressIdentifierProduct) {
        final Function function = new Function(
                FUNC_SETADDRESSASPRODUCTIDENTIFIER, 
                Arrays.asList(new Address(160, addressIdentifierProduct)), 
                Collections.emptyList());
        
        return FunctionEncoder.encode(function);
    }

    public static Function getContractValidationRules() {
        return new Function(FUNC_GETVALIDATIONRULESCONTRACTADDRESS, 
                Arrays.asList(), 
                Arrays.asList(new TypeReference<Address>() {}));

    }

    public static String approve(String spender, BigInteger amount) {
        final Function function = new Function(
                FUNC_APPROVE, 
                Arrays.asList(new Address(160, spender), 
                new Uint256(amount)), 
                Collections.emptyList());
        
        return FunctionEncoder.encode(function);
    }

    public static String burn(BigInteger amount) {
        final Function function = new Function(
                FUNC_BURN, 
                Arrays.asList(new Uint256(amount)), 
                Collections.emptyList());
                return FunctionEncoder.encode(function);
    }

    public static String burnFrom(String account, BigInteger amount) {
        final Function function = new Function(
                FUNC_BURNFROM, 
                Arrays.asList(new Address(160, account), 
                new Uint256(amount)), 
                Collections.emptyList());
                return FunctionEncoder.encode(function);
    }

}
