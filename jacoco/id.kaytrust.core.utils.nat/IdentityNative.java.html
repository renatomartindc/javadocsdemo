package id.kaytrust.core.utils.nat;

import java.io.IOException;
import java.math.BigInteger;

import org.web3j.protocol.exceptions.TransactionException;

import id.kaytrust.core.utils.exception.CapabilityException;
import id.kaytrust.core.utils.exception.ContractException;

public interface IdentityNative {

    public static IdentityNative build(String host, String token, String contractAddress, int blockTime) {
        return new IdentityNativeImpl(host, token, contractAddress, blockTime);
    }

    boolean hasCap(String addressFrom, String proxy, String device, String cap) throws IOException;

    String forward(String privateKey, String proxy, String addressContractDestination, String encodedFunction,
            double factor) throws IOException, TransactionException, CapabilityException ;

    String setCap(String privateKey, String proxy, String device, String cap, BigInteger startDate, BigInteger endDate,
            double factor) throws IOException, TransactionException, CapabilityException;

    String setDevice(String privateKey, String proxy, String device, BigInteger startDate,
            BigInteger endDate, double factor) throws IOException, TransactionException, CapabilityException;
            
    String createIdentity(String privateKey, double factor) throws IOException, TransactionException;
    
    boolean registeredIdentities(String addressFrom, String proxy) throws IOException;

    String getNetVersion() throws IOException;

    String deploySmartContract(String privateKey) throws IOException, ContractException;

    void shutdown();

}
