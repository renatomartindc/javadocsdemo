package id.kaytrust.core.utils.model.did;

import lombok.Getter;
import lombok.Setter;

/**
 * According To W3C the Did is the decentralized identifier that enables
 * verifiable, decentralized digital identity
 * <p>
 * . A DID identifies any subject (e.g., a person, organization, thing, data
 * model, abstract entity, etc.) that the controller of the DID decides that it
 * identifies Example:
 * 
 * <pre>
 * {@code
 * did:example:123456789abcdefghi
 * }
 * </pre>
 * 
 * The composing parts of the did are:
 * <ul>
 * <li>URL scheme identifier (did)
 * </ul>
 * <ul>
 * <li>Identifier for the DID method (example)
 * </ul>
 * <ul>
 * <li>DID method-specific identifier (123456789abcdefghi)
 * </ul>
 * Source: <a href="https://www.w3.org/TR/did-core/"></a>
 */
@Getter
@Setter
public class Did {

    private static final String SCHEME_IDENTIFIER = "did";
    /**
     * Identifier for the DID method
     */

    private String identifier;
    /**
     * DID method-specific identifier.
     */
    private String specificIdentifier;

    @Override
    public String toString() {
        return SCHEME_IDENTIFIER + ":" + getIdentifier() + ":" + getSpecificIdentifier();
    }
}
