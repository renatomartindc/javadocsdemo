package id.kaytrust.core.credential.ethereum.admin;

import id.kaytrust.core.credential.base.CredentialIssuer;
import id.kaytrust.core.utils.exception.ParamEmptyException;

/**
 * The generalized interface compliant with any ethereum network that considers the methods to issue a verifiable credential
 * in any ethereum network.
 * @see CredentialIssuer
 */
public interface EthereumCredentialIssuer extends CredentialIssuer {

	/**
	 * Sets an instance to interact with a verification registry smart contract. If the "contractAddress" argument is null
	 * a new Verification Registry smart contract is deployed on the ethereum network that is accessed through the "hostRPC".
	 * @param hostRpc                        host and port where ethereum node serves RPC requests
	 * @param token           				 can be blank, is token field Authorization
	 * @param privateKey                     The private key used to interact with the smart contract. In smart contracts it is the msg.sender
	 *                                       In this context this private key represents the issuer's private key.
	 * @param contractAddressIdentity        contract address corresponding to the Identity Manager contract to connect to.
	 *                                       that are to be queried.
	 * @param contractAddressCredential		 contract address VerifibleCredential
	 * @return instance EthereumCredentialIssuer
	 * @throws ParamEmptyException         if a required parameter is empty.
	 */
    static EthereumCredentialIssuer instance(String hostRpc, String token,String privateKey, String contractAddressIdentity, String contractAddressCredential) throws ParamEmptyException {
		return new EthereumCredentialIssuerImpl(hostRpc, token, privateKey, contractAddressIdentity, contractAddressCredential);
    }

		/**
	 * Sets an instance to interact with a verification registry smart contract. If the "contractAddress" argument is null
	 * a new Verification Registry smart contract is deployed on the ethereum network that is accessed through the "hostRPC".
	 * @param hostRpc                        host and port where ethereum node serves RPC requests
	 * @param token           				 can be blank, is token field Authorization
	 * @param privateKey                     The private key used to interact with the smart contract. In smart contracts it is the msg.sender
	 *                                       In this context this private key represents the issuer's private key.
	 * @param contractAddressIdentity        contract address corresponding to the Identity Manager contract to connect to.
	 *                                       that are to be queried.
	 * @param contractAddressCredential		 contract address VerifibleCredential
	 * @param blockTime 	  time mining of block in milliseconds, it's optional by default it's 15000
	 * @return instance EthereumCredentialIssuer
	 * @throws ParamEmptyException         if a required parameter is empty.
	 */
    static EthereumCredentialIssuer instance(String hostRpc, String token,String privateKey, String contractAddressIdentity, String contractAddressCredential, int blockTime) throws ParamEmptyException {
		return new EthereumCredentialIssuerImpl(hostRpc, token, privateKey, contractAddressIdentity, contractAddressCredential, blockTime);
    }

	
}
